
Sub uno(nombreArchivo As String)

Range("C1").value = "Categoría"
Range("D1").value = "COD SAP"
Range("E1").value = "Razón Social"
Range("F1").value = "Pedido"
Range("G1").value = "Vehículo"
Range("H1").value = "Placa"
Range("I1").value = "Comentario Cemento"
Range("J1").value = "MP RUC"
Range("K1").value = "MP Razón Social"
Range("L1").value = "Comentario Materia Prima"

ultimafila = ActiveSheet.UsedRange.SpecialCells(xlCellTypeLastCell).Row
Dim celda As Range
For Each celda In Range("B1:B" & ultimafila)
    If nombreArchivo = celda.value then
       celda.Offset(0,1).value = "No cumple cumple con los parámetros"
    End If
Next celda
End Sub
