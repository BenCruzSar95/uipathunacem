Sub ExcelDiario()
   ultimafila = ActiveSheet.UsedRange.SpecialCells(xlCellTypeLastCell).Row
   
    Columns("A:A").Select
    Selection.Insert Shift:=xlToRight, CopyOrigin:=xlFormatFromLeftOrAbove
    Range("A1").Select
    ActiveCell.FormulaR1C1 = "Doc"
    Columns("B:B").Select
    Selection.Insert Shift:=xlToRight, CopyOrigin:=xlFormatFromLeftOrAbove
    Range("B1").Select
    ActiveCell.FormulaR1C1 = "Centro"
    
    
    Dim celda As Range
    ultimafila = ActiveSheet.UsedRange.SpecialCells(xlCellTypeLastCell).Row
    For Each celda In Range("B2:B" & ultimafila)
        celda.Value = "UE01"
    Next celda
    
    Range("C1").Select
    ActiveCell.FormulaR1C1 = "F.Entrega"
    Columns("D:D").Select
    Selection.Insert Shift:=xlToRight, CopyOrigin:=xlFormatFromLeftOrAbove
    Range("D1").Select
    ActiveCell.FormulaR1C1 = "H.Entrega"
    Range("D1").Select
    
    Columns("E:E").Select
    Selection.Insert Shift:=xlToRight, CopyOrigin:=xlFormatFromLeftOrAbove
    Range("E1").Select
    ActiveCell.FormulaR1C1 = "Guía de Remisión"
    Range("E2").Select
    Columns("J:J").Select
    Selection.Copy
    Columns("F:F").Select
    Selection.Insert Shift:=xlToRight
    Application.CutCopyMode = False
    Columns("K:K").Select
    Selection.Delete Shift:=xlToLeft
    Columns("J:J").Select
    Selection.Copy
    Columns("G:G").Select
    Selection.Insert Shift:=xlToRight
    Application.CutCopyMode = False
    Columns("K:K").Select
    Selection.Delete Shift:=xlToLeft
    Range("G1").Select
    
    '
    Columns("H:H").Select
    Selection.Insert Shift:=xlToRight, CopyOrigin:=xlFormatFromLeftOrAbove
    Range("H1").Select
    ActiveCell.FormulaR1C1 = "Precio Neto"
    Range("H2").Select
    
    Columns("Y:Y").Select
    Selection.Copy
   
    Columns("I:I").Select
    Selection.Insert Shift:=xlToRight
    Application.CutCopyMode = False
    Range("I1").Select
    
    Columns("Z:Z").Select
    Selection.Delete Shift:=xlToLeft
    
    Columns("J:J").Select
    Selection.Insert Shift:=xlToRight, CopyOrigin:=xlFormatFromLeftOrAbove
    Range("J1").Select
    ActiveCell.FormulaR1C1 = "Clase doc.ventas"
    Range("J1").Select

End Sub
